bool recy_p(int A[],int p,int n)
{
    //验证p的合理性
    if(p<0 || p>=n) return false;
    //保存temp元素
    int temp_num=0;
    int index_move=0;
    //遍历整个数组
    for(int i=0;i<n;i++)
    {
        if((i-p)<0) index_move=n-p;
        else index_move=i-p;
        temp_num=A[index_move];
        A[index_move]=A[i];
        A[index_move]=temp_num;
    }
    return true;
} ^AEWHVkow
